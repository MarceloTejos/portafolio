
CREATE TABLE PAIS(

);

    MUN_ID INT, 
    MUN_ANO INT, 
    PAI_ID INT 
); 
ALTER TABLE MUNDIAL ADD CONSTRAINT MUNDIAL_PK PRIMARY KEY (MUN_ID); 

CREATE TABLE SELECCION( 
    SEL_ID INT, 
    SEL_NOMBRE VARCHAR(50), 
    MUN_ID INT
); 
ALTER TABLE SELECCION ADD CONSTRAINT SELECCION_PK PRIMARY KEY (SEL_ID); 

CREATE TABLE CIUDAD( 
    CIU_ID INT, 
    CIU_NOMBRE VARCHAR(50), 
    PAI_ID INT
); 
ALTER TABLE CIUDAD ADD CONSTRAINT CIUDAD_PK PRIMARY KEY (CIU_ID); 

CREATE TABLE ESTADIO( 
    EST_ID INT, 
    EST_NOMBRE VARCHAR(50), 
    CIU_ID INT
); 
ALTER TABLE ESTADIO ADD CONSTRAINT ESTADIO_PK PRIMARY KEY (EST_ID); 

CREATE TABLE ARBITRO( 
    ARB_ID INTEGER NOT NULL, 
    ARB_NOMBRE VARCHAR(30) NOT NULL 
);
ALTER TABLE ARBITRO ADD CONSTRAINT ARBITRO_PK PRIMARY KEY (ARB_ID); 

CREATE TABLE FUNCION_ARBITRO( 
    FAR_ID INT, 
    FAR_NOMBRE VARCHAR(30)
);
ALTER TABLE FUNCION_ARBITRO ADD CONSTRAINT FUNCION_ARBITRO_PK PRIMARY KEY (FAR_ID); 

CREATE TABLE POSICION( 
    POS_ID INTEGER NOT NULL, 
    POS_NOMBRE VARCHAR(50) NOT NULL 
); 
ALTER TABLE POSICION ADD CONSTRAINT POSICION_PK PRIMARY KEY (POS_ID); 

CREATE TABLE JUGADOR( 
    JUG_ID INT, 
    JUG_NOMBRE VARCHAR(50), 
    SEL_ID INT
);
ALTER TABLE JUGADOR ADD CONSTRAINT JUGADOR_PK PRIMARY KEY (JUG_ID);

CREATE TABLE SANCION( 
    SAN_ID INT, 
    SAN_NOMBRE VARCHAR(50), 
    JUG_ID INT
); 
ALTER TABLE SANCION ADD CONSTRAINT SANCION_PK PRIMARY KEY (SAN_ID);

CREATE TABLE EQUIPO_TECNICO( 
    EQT_ID INT, 
    EQT_NOMBRE VARCHAR(50), 
    SEL_ID INT 
); 
ALTER TABLE EQUIPO_TECNICO ADD CONSTRAINT EQUIPO_TECNICO_PK PRIMARY KEY (EQT_ID); 

CREATE TABLE PARTIDO( 
    PAR_ID INT, 
    PAR_FECHA DATE,  
    PAR_GRUPO VARCHAR(1), 
    EST_ID INT, 
    SEL_ID INT, 
    SEL_IDV INT 
); 
ALTER TABLE PARTIDO ADD CONSTRAINT PARTIDO_PK PRIMARY KEY (PAR_ID); 

CREATE TABLE ALINEACION ( 
    ALI_ID INT, 
    PAR_ID  INT, 
    POS_ID INT, 
    JUG_ID  INT 
); 
ALTER TABLE ALINEACION ADD CONSTRAINT ALINEACION_PK PRIMARY KEY (ALI_ID); 

CREATE TABLE ARBITRO_PARTIDO( 
    ARP_ID INT, 
    ARP_NOMBRE VARCHAR(30), 
    PAR_ID INT, 
    ARB_ID INT, 
    FAR_ID INT
); 
ALTER TABLE ARBITRO_PARTIDO ADD CONSTRAINT ARBITRO_PARTIDO_PK PRIMARY KEY (ARP_ID); 

CREATE TABLE DETALLE_PARTIDO( 
    DPA_ID INT, 
    DPA_MINUTO INT, 
    PAR_ID INT, 
    SAN_ID INT,
    JUG_ID INT,
    SEL_ID INT
);
